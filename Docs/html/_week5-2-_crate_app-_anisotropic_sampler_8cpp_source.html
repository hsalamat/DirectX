185bool CrateApp::Initialize()
186{
187 if(!D3DApp::Initialize())
188 return false;
189
190 // Reset the command list to prep for initialization commands.
191 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
192
193 // Get the increment size of a descriptor in this heap type. This is hardware specific,
194 // so we have to query this information.
195 mCbvSrvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
196
197 LoadTextures();
198
199 BuildRootSignature();
200 BuildDescriptorHeaps();
201
202 BuildShadersAndInputLayout();
203 BuildShapeGeometry();
204 BuildMaterials();
205 BuildRenderItems();
206 BuildFrameResources();
207 BuildPSOs();
208
209 // Execute the initialization commands.
210 ThrowIfFailed(mCommandList->Close());
211 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
212 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
213
214 // Wait until initialization is complete.
215 FlushCommandQueue();
216
217 return true;
218}

229void CrateApp::Update(const GameTimer& gt)
230{
231 OnKeyboardInput(gt);
232 UpdateCamera(gt);
233
234 // Cycle through the circular frame resource array.
235 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
236 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
237
238 // Has the GPU finished processing the commands of the current frame resource?
239 // If not, wait until the GPU has completed commands up to this fence point.
240 if(mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
241 {
242 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
243 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
244 WaitForSingleObject(eventHandle, INFINITE);
245 CloseHandle(eventHandle);
246 }
247
248 AnimateMaterials(gt);
249 UpdateObjectCBs(gt);
250 UpdateMaterialCBs(gt);
251 UpdateMainPassCB(gt);

254void CrateApp::Draw(const GameTimer& gt)
255{
256 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
257
258 // Reuse the memory associated with command recording.
259 // We can only reset when the associated command lists have finished execution on the GPU.
260 ThrowIfFailed(cmdListAlloc->Reset());
261
262 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
263 // Reusing the command list reuses memory.
264 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mOpaquePSO.Get()));
265
266 mCommandList->RSSetViewports(1, &mScreenViewport);
267 mCommandList->RSSetScissorRects(1, &mScissorRect);
268
269 // Indicate a state transition on the resource usage.
270 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
271 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
272
273 // Clear the back buffer and depth buffer.
274 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
275 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
276
277 // Specify the buffers we are going to render to.
278 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
279
280 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
281 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
282
283 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
284
285 auto passCB = mCurrFrameResource->PassCB->Resource();
286 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
287
288 DrawRenderItems(mCommandList.Get(), mOpaqueRitems);
289
290 // Indicate a state transition on the resource usage.
291 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
292 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
293
294 // Done recording commands.
295 ThrowIfFailed(mCommandList->Close());
296
297 // Add the command list to the queue for execution.
298 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
299 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
300
301 // Swap the back and front buffers
302 ThrowIfFailed(mSwapChain->Present(0, 0));
303 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
304
305 // Advance the fence value to mark commands up to this fence point.
306 mCurrFrameResource->Fence = ++mCurrentFence;
307
308 // Add an instruction to the command queue to set a new fence point.
309 // Because we are on the GPU timeline, the new fence point won't be
310 // set until the GPU finishes processing all the commands prior to this Signal().
311 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
312}

327void CrateApp::OnMouseMove(WPARAM btnState, int x, int y)
328{
329 if((btnState & MK_LBUTTON) != 0)
330 {
331 // Make each pixel correspond to a quarter of a degree.
332 float dx = XMConvertToRadians(0.25f*static_cast<float>(x - mLastMousePos.x));
333 float dy = XMConvertToRadians(0.25f*static_cast<float>(y - mLastMousePos.y));
334
335 // Update angles based on input to orbit camera around box.
336 mTheta += dx;
337 mPhi += dy;
338
339 // Restrict the angle mPhi.
340 mPhi = MathHelper::Clamp(mPhi, 0.1f, MathHelper::Pi - 0.1f);
341 }
342 else if((btnState & MK_RBUTTON) != 0)
343 {
344 // Make each pixel correspond to 0.2 unit in the scene.
345 float dx = 0.05f*static_cast<float>(x - mLastMousePos.x);
346 float dy = 0.05f*static_cast<float>(y - mLastMousePos.y);
347
348 // Update the camera radius based on input.
349 mRadius += dx - dy;
350
351 // Restrict the radius.
352 mRadius = MathHelper::Clamp(mRadius, 5.0f, 150.0f);
353 }
354
355 mLastMousePos.x = x;
356 mLastMousePos.y = y;
357}

363void CrateApp::UpdateCamera(const GameTimer& gt)
364{
365 // Convert Spherical to Cartesian coordinates.
366 mEyePos.x = mRadius*sinf(mPhi)*cosf(mTheta);
367 mEyePos.z = mRadius*sinf(mPhi)*sinf(mTheta);
368 mEyePos.y = mRadius*cosf(mPhi);
369
370 // Build the view matrix.
371 XMVECTOR pos = XMVectorSet(mEyePos.x, mEyePos.y, mEyePos.z, 1.0f);
372 XMVECTOR target = XMVectorZero();
373 XMVECTOR up = XMVectorSet(0.0f, 1.0f, 0.0f, 0.0f);
374
375 XMMATRIX view = XMMatrixLookAtLH(pos, target, up);
376 XMStoreFloat4x4(&mView, view);
377}

384void CrateApp::UpdateObjectCBs(const GameTimer& gt)
385{
386 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
387 for(auto& e : mAllRitems)
388 {
389 // Only update the cbuffer data if the constants have changed.
390 // This needs to be tracked per frame resource.
391 if(e->NumFramesDirty > 0)
392 {
393 XMMATRIX world = XMLoadFloat4x4(&e->World);
394 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
395
396 ObjectConstants objConstants;
397 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
398 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
399
400 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
401
402 // Next FrameResource need to be updated too.
403 e->NumFramesDirty--;
404 }
405 }
406}

408void CrateApp::UpdateMaterialCBs(const GameTimer& gt)
409{
410 auto currMaterialCB = mCurrFrameResource->MaterialCB.get();
411 for(auto& e : mMaterials)
412 {
413 // Only update the cbuffer data if the constants have changed. If the cbuffer
414 // data changes, it needs to be updated for each FrameResource.
415 Material* mat = e.second.get();
416 if(mat->NumFramesDirty > 0)
417 {
418 XMMATRIX matTransform = XMLoadFloat4x4(&mat->MatTransform);
419
420 MaterialConstants matConstants;
421 matConstants.DiffuseAlbedo = mat->DiffuseAlbedo;
422 matConstants.FresnelR0 = mat->FresnelR0;
423 matConstants.Roughness = mat->Roughness;
424 XMStoreFloat4x4(&matConstants.MatTransform, XMMatrixTranspose(matTransform));
425
426 currMaterialCB->CopyData(mat->MatCBIndex, matConstants);
427
428 // Next FrameResource need to be updated too.
429 mat->NumFramesDirty--;
430 }

434void CrateApp::UpdateMainPassCB(const GameTimer& gt)
435{
436 XMMATRIX view = XMLoadFloat4x4(&mView);
437 XMMATRIX proj = XMLoadFloat4x4(&mProj);
438
439 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
440 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
441 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
442 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
443
444 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
445 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
446 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
447 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
448 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
449 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
450 mMainPassCB.EyePosW = mEyePos;
451 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
452 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
453 mMainPassCB.NearZ = 1.0f;
454 mMainPassCB.FarZ = 1000.0f;
455 mMainPassCB.TotalTime = gt.TotalTime();
456 mMainPassCB.DeltaTime = gt.DeltaTime();
457 mMainPassCB.AmbientLight = { 0.25f, 0.25f, 0.35f, 1.0f };
458 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
459 mMainPassCB.Lights[0].Strength = { 0.6f, 0.6f, 0.6f };
460 mMainPassCB.Lights[1].Direction = { -0.57735f, -0.57735f, 0.57735f };
461 mMainPassCB.Lights[1].Strength = { 0.3f, 0.3f, 0.3f };
462 mMainPassCB.Lights[2].Direction = { 0.0f, -0.707f, -0.707f };
463 mMainPassCB.Lights[2].Strength = { 0.15f, 0.15f, 0.15f };
464
465 auto currPassCB = mCurrFrameResource->PassCB.get();
466 currPassCB->CopyData(0, mMainPassCB);
467}

481void CrateApp::BuildRootSignature()
482{
483 CD3DX12_DESCRIPTOR_RANGE texTable;
484 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
485
486 // Root parameter can be a table, root descriptor or root constants.
487 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
488
489 // Perfomance TIP: Order from most frequent to least frequent.
490 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
491 slotRootParameter[1].InitAsConstantBufferView(0);
492 slotRootParameter[2].InitAsConstantBufferView(1);
493 slotRootParameter[3].InitAsConstantBufferView(2);
494
495 auto staticSamplers = GetStaticSamplers();
496
497 // A root signature is an array of root parameters.
498 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
499 //define an array of so - called static samplers your application can use.
500 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
501 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
502 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
503
504 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
505 ComPtr<ID3DBlob> serializedRootSig = nullptr;
506 ComPtr<ID3DBlob> errorBlob = nullptr;
507 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
508 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
509
510 if(errorBlob != nullptr)
511 {
512 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
513 }
514 ThrowIfFailed(hr);
515
516 ThrowIfFailed(md3dDevice->CreateRootSignature(
517 0,
518 serializedRootSig->GetBufferPointer(),
519 serializedRootSig->GetBufferSize(),
520 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
521}

523//Once a texture resource is created, we need to create an SRV descriptor to it which we
524//can set to a root signature parameter slot for use by the shader programs.
525void CrateApp::BuildDescriptorHeaps()
526{
527 //
528 // Create the SRV heap.
529 //
530 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
531 srvHeapDesc.NumDescriptors = 1;
532 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
533 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
534 ThrowIfFailed(md3dDevice->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&mSrvDescriptorHeap)));
535
536 //
537 // Fill out the heap with actual descriptors.
538 //
539 CD3DX12_CPU_DESCRIPTOR_HANDLE hDescriptor(mSrvDescriptorHeap->GetCPUDescriptorHandleForHeapStart());
540
541 auto woodCrateTex = mTextures["woodCrateTex"]->Resource;
542
543 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
544
545 //This mapping enables the shader resource view (SRV) to choose how memory gets routed to the 4 return components in a shader after a memory fetch.
546 //When a texture is sampled in a shader, it will return a vector of the texture data at the specified texture coordinates.
547 //This field provides a way to reorder the vector components returned when sampling the texture.
548 //D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING will not reorder the components and just return the data in the order it is stored in the texture resource.
549 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
550
551 srvDesc.Format = woodCrateTex->GetDesc().Format;
552 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
553 srvDesc.Texture2D.MostDetailedMip = 0;
554 //The number of mipmap levels to view, starting at MostDetailedMip.This field, along with MostDetailedMip allows us to
555 //specify a subrange of mipmap levels to view.You can specify - 1 to indicate to view
556 //all mipmap levels from MostDetailedMip down to the last mipmap level.
557
558 srvDesc.Texture2D.MipLevels = woodCrateTex->GetDesc().MipLevels;
559
560 //Specifies the minimum mipmap level that can be accessed. 0.0 means all the mipmap levels can be accessed.
561 //Specifying 3.0 means mipmap levels 3.0 to MipCount - 1 can be accessed.
562 srvDesc.Texture2D.ResourceMinLODClamp = 0.0f;
563
564 md3dDevice->CreateShaderResourceView(woodCrateTex.Get(), &srvDesc, hDescriptor);
565}

580void CrateApp::BuildShapeGeometry()
581{
582 GeometryGenerator geoGen;
583 GeometryGenerator::MeshData box = geoGen.CreateBox(1.0f, 1.0f, 1.0f, 3);
584
585 SubmeshGeometry boxSubmesh;
586 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
587 boxSubmesh.StartIndexLocation = 0;
588 boxSubmesh.BaseVertexLocation = 0;
589
590
591 std::vector<Vertex> vertices(box.Vertices.size());
592
593 for(size_t i = 0; i < box.Vertices.size(); ++i)
594 {
595 vertices[i].Pos = box.Vertices[i].Position;
596 vertices[i].Normal = box.Vertices[i].Normal;
597 vertices[i].TexC = box.Vertices[i].TexC;
598 }
599
600 std::vector<std::uint16_t> indices = box.GetIndices16();
601
602 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
603 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
604
605 auto geo = std::make_unique<MeshGeometry>();
606 geo->Name = "boxGeo";
607
608 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
609 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
610
611 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
612 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
613
614 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
615 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
616
617 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
618 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
619

702void CrateApp::DrawRenderItems(ID3D12GraphicsCommandList* cmdList, const std::vector<RenderItem*>& ritems)
703{
704 UINT objCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(ObjectConstants));
705 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
706
707 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
708 auto matCB = mCurrFrameResource->MaterialCB->Resource();
709
710 // For each render item...
711 for(size_t i = 0; i < ritems.size(); ++i)
712 {
713 auto ri = ritems[i];
714
715 cmdList->IASetVertexBuffers(0, 1, &ri->Geo->VertexBufferView());
716 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
717 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
718
719 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
720 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
721
722 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex*objCBByteSize;
723 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex*matCBByteSize;
724
725 cmdList->SetGraphicsRootDescriptorTable(0, tex);
726 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
727 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
728
729 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
730 }

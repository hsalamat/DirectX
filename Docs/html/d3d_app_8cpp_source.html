72int D3DApp::Run()
73{
74 MSG msg = {0};
75
76 mTimer.Reset();
77
78 while(msg.message != WM_QUIT)
79 {
80 // If there are Window messages then process them.
81 if(PeekMessage( &msg, 0, 0, 0, PM_REMOVE ))
82 {
83 TranslateMessage( &msg );
84 DispatchMessage( &msg );
85 }
86 // Otherwise, do animation/game stuff.
87 else
88 {
89 mTimer.Tick();
90
91 if( !mAppPaused )
92 {
93 CalculateFrameStats();
94 Update(mTimer);
95 Draw(mTimer);
96 }
97 else
98 {
99 Sleep(100);
100 }
101 }
102 }
103
104 return (int)msg.wParam;
105}

237LRESULT D3DApp::MsgProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
238{
239 switch( msg )
240 {
244 case WM_ACTIVATE:
245 if( LOWORD(wParam) == WA_INACTIVE )
246 {
247 mAppPaused = true;
248 mTimer.Stop();
249 }
250 else
251 {
252 mAppPaused = false;
253 mTimer.Start();
254 }
255 return 0;
256
258 case WM_SIZE:
260 mClientWidth = LOWORD(lParam);
261 mClientHeight = HIWORD(lParam);
262 if( md3dDevice )
263 {
264 if( wParam == SIZE_MINIMIZED )
265 {
266 mAppPaused = true;
267 mMinimized = true;
268 mMaximized = false;
269 }
270 else if( wParam == SIZE_MAXIMIZED )
271 {
272 mAppPaused = false;
273 mMinimized = false;
274 mMaximized = true;
275 OnResize();
276 }
277 else if( wParam == SIZE_RESTORED )
278 {
279
280 // Restoring from minimized state?
281 if( mMinimized )
282 {
283 mAppPaused = false;
284 mMinimized = false;
285 OnResize();
286 }
287
288 // Restoring from maximized state?
289 else if( mMaximized )
290 {
291 mAppPaused = false;
292 mMaximized = false;
293 OnResize();
294 }
295 else if( mResizing )
296 {
305 }
306 else // API call such as SetWindowPos or mSwapChain->SetFullscreenState.
307 {
308 OnResize();
309 }
310 }
311 }
312 return 0;
313
315 case WM_ENTERSIZEMOVE:
316 mAppPaused = true;
317 mResizing = true;
318 mTimer.Stop();
319 return 0;
320
323 case WM_EXITSIZEMOVE:
324 mAppPaused = false;
325 mResizing = false;
326 mTimer.Start();
327 OnResize();
328 return 0;
329
331 case WM_DESTROY:
332 PostQuitMessage(0);
333 return 0;
334
337 case WM_MENUCHAR:
338 // Don't beep when we alt-enter.
339 return MAKELRESULT(0, MNC_CLOSE);
340
342 case WM_GETMINMAXINFO:
343 ((MINMAXINFO*)lParam)->ptMinTrackSize.x = 200;
344 ((MINMAXINFO*)lParam)->ptMinTrackSize.y = 200;
345 return 0;
346
347 case WM_LBUTTONDOWN:
348 case WM_MBUTTONDOWN:
349 case WM_RBUTTONDOWN:
350 OnMouseDown(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
351 return 0;
352 case WM_LBUTTONUP:
353 case WM_MBUTTONUP:
354 case WM_RBUTTONUP:
355 OnMouseUp(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
356 return 0;
357 case WM_MOUSEMOVE:
358 OnMouseMove(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
359 return 0;
360 case WM_KEYUP:
361 if(wParam == VK_ESCAPE)
362 {
363 PostQuitMessage(0);
364 }
365 else if((int)wParam == VK_F2)
366 Set4xMsaaState(!m4xMsaaState);
367
368 return 0;
369 }
370
371 return DefWindowProc(hwnd, msg, wParam, lParam);
372}

555void D3DApp::FlushCommandQueue()
556{
558 mCurrentFence++;
559
563 ThrowIfFailed(mCommandQueue->Signal(mFence.Get(), mCurrentFence));
564
566 if(mFence->GetCompletedValue() < mCurrentFence)
567 {
568 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
569
571 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrentFence, eventHandle));
572
574 WaitForSingleObject(eventHandle, INFINITE);
575 CloseHandle(eventHandle);
576 }
577}

599void D3DApp::CalculateFrameStats()
600{
601 // Code computes the average frames per second, and also the
602 // average time it takes to render one frame. These stats
603 // are appended to the window caption bar.
604
605 static int frameCnt = 0;
606 static float timeElapsed = 0.0f;
607
608 frameCnt++;
609
610 // Compute averages over one second period.
611 if( (mTimer.TotalTime() - timeElapsed) >= 1.0f )
612 {
613 float fps = (float)frameCnt; // fps = frameCnt / 1
614 float mspf = 1000.0f / fps;
615
616 wstring fpsStr = to_wstring(fps);
617 wstring mspfStr = to_wstring(mspf);
618
619 wstring windowText = mMainWndCaption +
620 L" fps: " + fpsStr +
621 L" mspf: " + mspfStr;
622
623 SetWindowText(mhMainWnd, windowText.c_str());
624
625 // Reset for next average.
626 frameCnt = 0;
627 timeElapsed += 1.0f;
628 }
629}

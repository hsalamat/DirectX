265void CrateApp::Draw(const GameTimer& gt)
266{
267 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
268
269 // Reuse the memory associated with command recording.
270 // We can only reset when the associated command lists have finished execution on the GPU.
271 ThrowIfFailed(cmdListAlloc->Reset());
272
273 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
274 // Reusing the command list reuses memory.
275 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mOpaquePSO.Get()));
276
277 mCommandList->RSSetViewports(1, &mScreenViewport);
278 mCommandList->RSSetScissorRects(1, &mScissorRect);
279
280 // Indicate a state transition on the resource usage.
281 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
282 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
283
284 // Clear the back buffer and depth buffer.
285 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
286 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
287
288 // Specify the buffers we are going to render to.
289 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
290
291 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
292 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
293
294 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
295
296 auto passCB = mCurrFrameResource->PassCB->Resource();
297 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
298
299 DrawRenderItems(mCommandList.Get(), mOpaqueRitems);
300
301 // Indicate a state transition on the resource usage.
302 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
303 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
304
305 // Done recording commands.
306 ThrowIfFailed(mCommandList->Close());
307
308 // Add the command list to the queue for execution.
309 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
310 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
311
312 // Swap the back and front buffers
313 ThrowIfFailed(mSwapChain->Present(0, 0));
314 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
315
316 // Advance the fence value to mark commands up to this fence point.
317 mCurrFrameResource->Fence = ++mCurrentFence;
318
319 // Add an instruction to the command queue to set a new fence point.
320 // Because we are on the GPU timeline, the new fence point won't be
321 // set until the GPU finishes processing all the commands prior to this Signal().
322 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
323}

338void CrateApp::OnMouseMove(WPARAM btnState, int x, int y)
339{
340 if((btnState & MK_LBUTTON) != 0)
341 {
342 // Make each pixel correspond to a quarter of a degree.
343 float dx = XMConvertToRadians(0.25f*static_cast<float>(x - mLastMousePos.x));
344 float dy = XMConvertToRadians(0.25f*static_cast<float>(y - mLastMousePos.y));
345
346 // Update angles based on input to orbit camera around box.
347 mTheta += dx;
348 mPhi += dy;
349
350 // Restrict the angle mPhi.
351 mPhi = MathHelper::Clamp(mPhi, 0.1f, MathHelper::Pi - 0.1f);
352 }
353 else if((btnState & MK_RBUTTON) != 0)
354 {
355 // Make each pixel correspond to 0.2 unit in the scene.
356 float dx = 0.05f*static_cast<float>(x - mLastMousePos.x);
357 float dy = 0.05f*static_cast<float>(y - mLastMousePos.y);
358
359 // Update the camera radius based on input.
360 mRadius += dx - dy;
361
362 // Restrict the radius.
363 mRadius = MathHelper::Clamp(mRadius, 5.0f, 150.0f);
364 }
365
366 mLastMousePos.x = x;
367 mLastMousePos.y = y;
368}

374void CrateApp::UpdateCamera(const GameTimer& gt)
375{
376 // Convert Spherical to Cartesian coordinates.
377 mEyePos.x = mRadius*sinf(mPhi)*cosf(mTheta);
378 mEyePos.z = mRadius*sinf(mPhi)*sinf(mTheta);
379 mEyePos.y = mRadius*cosf(mPhi);
380
381 // Build the view matrix.
382 XMVECTOR pos = XMVectorSet(mEyePos.x, mEyePos.y, mEyePos.z, 1.0f);
383 XMVECTOR target = XMVectorZero();
384 XMVECTOR up = XMVectorSet(0.0f, 1.0f, 0.0f, 0.0f);
385
386 XMMATRIX view = XMMatrixLookAtLH(pos, target, up);
387 XMStoreFloat4x4(&mView, view);
388}

395void CrateApp::UpdateObjectCBs(const GameTimer& gt)
396{
397 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
398 for(auto& e : mAllRitems)
399 {
400 // Only update the cbuffer data if the constants have changed.
401 // This needs to be tracked per frame resource.
402 if(e->NumFramesDirty > 0)
403 {
404 XMMATRIX world = XMLoadFloat4x4(&e->World);
405 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
406
407 ObjectConstants objConstants;
408 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
409 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
410
411 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
412
413 // Next FrameResource need to be updated too.
414 e->NumFramesDirty--;
415 }
416 }

445void CrateApp::UpdateMainPassCB(const GameTimer& gt)
446{
447 XMMATRIX view = XMLoadFloat4x4(&mView);
448 XMMATRIX proj = XMLoadFloat4x4(&mProj);
449
450 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
451 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
452 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
453 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
454
455 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
456 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
457 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
458 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
459 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
460 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
461 mMainPassCB.EyePosW = mEyePos;
462 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);

467 mMainPassCB.DeltaTime = gt.DeltaTime();

493void CrateApp::BuildRootSignature()
494{
495 //step22
496 CD3DX12_DESCRIPTOR_RANGE texTable;
497 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
498
499 // Root parameter can be a table, root descriptor or root constants.
500 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
501
502 // Perfomance TIP: Order from most frequent to least frequent.
503 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
504 slotRootParameter[1].InitAsConstantBufferView(0);
505 slotRootParameter[2].InitAsConstantBufferView(1);
506 slotRootParameter[3].InitAsConstantBufferView(2);
507
508 auto staticSamplers = GetStaticSamplers();
509
510 // A root signature is an array of root parameters.
511 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
512 //define an array of so - called static samplers your application can use.
513 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
514 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
515 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
516
517 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
518 ComPtr<ID3DBlob> serializedRootSig = nullptr;
519 ComPtr<ID3DBlob> errorBlob = nullptr;
520 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
521 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
522
523 if(errorBlob != nullptr)
524 {
525 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
526 }
527 ThrowIfFailed(hr);
528
529 ThrowIfFailed(md3dDevice->CreateRootSignature(
530 0,
531 serializedRootSig->GetBufferPointer(),
532 serializedRootSig->GetBufferSize(),
533 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
534}

537//Once a texture resource is created, we need to create an SRV descriptor to it which we
538//can set to a root signature parameter slot for use by the shader programs.
539void CrateApp::BuildDescriptorHeaps()
540{
541 //
542 // Create the SRV heap.
543 //
544 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
545 srvHeapDesc.NumDescriptors = 1;
546 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
547 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
548 ThrowIfFailed(md3dDevice->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&mSrvDescriptorHeap)));
549
550 //
551 // Fill out the heap with actual descriptors.
552 //
553 CD3DX12_CPU_DESCRIPTOR_HANDLE hDescriptor(mSrvDescriptorHeap->GetCPUDescriptorHandleForHeapStart());
554
555 auto woodCrateTex = mTextures["woodCrateTex"]->Resource;
556
557 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
558
559 //This mapping enables the shader resource view (SRV) to choose how memory gets routed to the 4 return components in a shader after a memory fetch.
560 //When a texture is sampled in a shader, it will return a vector of the texture data at the specified texture coordinates.
561 //This field provides a way to reorder the vector components returned when sampling the texture.
562 //D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING will not reorder the components and just return the data in the order it is stored in the texture resource.
563 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
564
565 srvDesc.Format = woodCrateTex->GetDesc().Format;
566 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
567 srvDesc.Texture2D.MostDetailedMip = 0;
568 //The number of mipmap levels to view, starting at MostDetailedMip.This field, along with MostDetailedMip allows us to
569 //specify a subrange of mipmap levels to view.You can specify - 1 to indicate to view
570 //all mipmap levels from MostDetailedMip down to the last mipmap level.
571
572 srvDesc.Texture2D.MipLevels = woodCrateTex->GetDesc().MipLevels;
573
574 //Specifies the minimum mipmap level that can be accessed. 0.0 means all the mipmap levels can be accessed.
575 //Specifying 3.0 means mipmap levels 3.0 to MipCount - 1 can be accessed.
576 srvDesc.Texture2D.ResourceMinLODClamp = 0.0f;
577
578 md3dDevice->CreateShaderResourceView(woodCrateTex.Get(), &srvDesc, hDescriptor);
579}

596void CrateApp::BuildShapeGeometry()
597{
598 GeometryGenerator geoGen;
599 GeometryGenerator::MeshData box = geoGen.CreateBox(1.0f, 1.0f, 1.0f, 3);
600
601 SubmeshGeometry boxSubmesh;
602 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
603 boxSubmesh.StartIndexLocation = 0;
604 boxSubmesh.BaseVertexLocation = 0;
605
606
607 std::vector<Vertex> vertices(box.Vertices.size());
608
609 for(size_t i = 0; i < box.Vertices.size(); ++i)
610 {
611 vertices[i].Pos = box.Vertices[i].Position;
612 vertices[i].Normal = box.Vertices[i].Normal;
613 vertices[i].TexC = box.Vertices[i].TexC;
614 }
615
616 std::vector<std::uint16_t> indices = box.GetIndices16();
617
618 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
619 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
620
621 auto geo = std::make_unique<MeshGeometry>();
622 geo->Name = "boxGeo";
623
624 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
625 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
626
627 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
628 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
629
630 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),

722{
723 UINT objCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(ObjectConstants));
724 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
725
726 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
727 auto matCB = mCurrFrameResource->MaterialCB->Resource();
728
729 // For each render item...
730 for(size_t i = 0; i < ritems.size(); ++i)
731 {
732 auto ri = ritems[i];
733
734 cmdList->IASetVertexBuffers(0, 1, &ri->Geo->VertexBufferView());
735 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
736 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
737
738 //step18
739 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
740 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
741
742 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex*objCBByteSize;
743 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex*matCBByteSize;
744
745 //step19: assuming the root signature has been defined to expect a table of shader
746 //resource views to be bound to the 0th slot parameter
747 cmdList->SetGraphicsRootDescriptorTable(0, tex);
748 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
749 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
750
751 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
752 }

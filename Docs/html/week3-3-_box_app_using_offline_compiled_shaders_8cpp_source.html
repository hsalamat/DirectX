169void BoxApp::Update(const GameTimer& gt)
170{
171 // Convert Spherical to Cartesian coordinates.
172 float x = mRadius*sinf(mPhi)*cosf(mTheta);
173 float z = mRadius*sinf(mPhi)*sinf(mTheta);
174 float y = mRadius*cosf(mPhi);
175
176 // Build the view matrix.
177 XMVECTOR pos = XMVectorSet(x, y, z, 1.0f);
178 XMVECTOR target = XMVectorZero();
179 XMVECTOR up = XMVectorSet(0.0f, 1.0f, 0.0f, 0.0f);
180
181 XMMATRIX view = XMMatrixLookAtLH(pos, target, up);
182 XMStoreFloat4x4(&mView, view);
183
184
185
186
187 XMMATRIX world = XMLoadFloat4x4(&mWorld);
188 XMMATRIX proj = XMLoadFloat4x4(&mProj);
189 XMMATRIX worldViewProj = world*view*proj;
190
191 // Update the constant buffer with the latest worldViewProj matrix.
192 ObjectConstants objConstants;
193 XMStoreFloat4x4(&objConstants.WorldViewProj, XMMatrixTranspose(worldViewProj));
194 mObjectCB->CopyData(0, objConstants);
195
196}

198void BoxApp::Draw(const GameTimer& gt)
199{
200 // Reuse the memory associated with command recording.
201 // We can only reset when the associated command lists have finished execution on the GPU.
202 ThrowIfFailed(mDirectCmdListAlloc->Reset());
203
204 //step3: A command list can be reset after it has been added to the command queue via ExecuteCommandList.
205 // Reusing the command list reuses memory.
206 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), mPSO.Get()));
207
208 mCommandList->RSSetViewports(1, &mScreenViewport);
209 mCommandList->RSSetScissorRects(1, &mScissorRect);
210
211 // Indicate a state transition on the resource usage.
212 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
213 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
214
215
216 // Clear the back buffer and depth buffer.
217 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
218 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
219
220 // Specify the buffers we are going to render to.
221 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
222
223 ID3D12DescriptorHeap* descriptorHeaps[] = { mCbvHeap.Get() };
224 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
225
226 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
227
228 mCommandList->IASetVertexBuffers(0, 1, &mBoxGeo->VertexBufferView());
229 mCommandList->IASetIndexBuffer(&mBoxGeo->IndexBufferView());
230 mCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
231
232 mCommandList->SetGraphicsRootDescriptorTable(0, mCbvHeap->GetGPUDescriptorHandleForHeapStart());
233
234 mCommandList->DrawIndexedInstanced(
235 mBoxGeo->DrawArgs["box"].IndexCount,
236 1, 0, 0, 0);
237
238 // Indicate a state transition on the resource usage.
239 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
240 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
241
242 //step4: Done recording commands. Comment this line out to see what debugger will tell you in the output window.
243 ThrowIfFailed(mCommandList->Close());
244
245 // Add the command list to the queue for execution.
246 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
247 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
248
249 // swap the back and front buffers
250 //HRESULT Present(UINT SyncInterval,UINT Flags);
251 //SyncInterval: 0 - The presentation occurs immediately, there is no synchronization.
252 //Flags defined by the DXGI_PRESENT constants. 0: Present a frame from each buffer (starting with the current buffer) to the output.
253 ThrowIfFailed(mSwapChain->Present(0, 0));
254 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
255
256 // Wait until frame commands are complete. This waiting is inefficient and is
257 // done for simplicity. Later we will show how to organize our rendering code
258 // so we do not have to wait per frame.
259 FlushCommandQueue();
260}

277void BoxApp::OnMouseMove(WPARAM btnState, int x, int y)
278{
279 if((btnState & MK_LBUTTON) != 0)
280 {
281 // Make each pixel correspond to a quarter of a degree.
282 float dx = XMConvertToRadians(0.25f*static_cast<float>(x - mLastMousePos.x));
283 float dy = XMConvertToRadians(0.25f*static_cast<float>(y - mLastMousePos.y));
284
285 // Update angles based on input to orbit camera around box.
286 mTheta += dx;
287 mPhi += dy;
288
289 // Restrict the angle mPhi.
290 mPhi = MathHelper::Clamp(mPhi, 0.1f, MathHelper::Pi - 0.1f);
291 }
292 else if((btnState & MK_RBUTTON) != 0)
293 {
294 // Make each pixel correspond to 0.005 unit in the scene.
295 float dx = 0.005f*static_cast<float>(x - mLastMousePos.x);
296 float dy = 0.005f*static_cast<float>(y - mLastMousePos.y);
297
298 // Update the camera radius based on input.
299 mRadius += dx - dy;
300
301 // Restrict the radius.
302 mRadius = MathHelper::Clamp(mRadius, 3.0f, 15.0f);
303 }
304
305 mLastMousePos.x = x;
306 mLastMousePos.y = y;
307}

342void BoxApp::BuildRootSignature()
343{
344 // Shader programs typically require resources as input (constant buffers,
345 // textures, samplers). The root signature defines the resources the shader
346 // programs expect. If we think of the shader programs as a function, and
347 // the input resources as function parameters, then the root signature can be
348 // thought of as defining the function signature.
349
350 // Root parameter can be a table, root descriptor or root constants.
351 CD3DX12_ROOT_PARAMETER slotRootParameter[1];
352
353 // Create a single descriptor table of CBVs.
354 CD3DX12_DESCRIPTOR_RANGE cbvTable;
355 cbvTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_CBV, 1, 0);
356 slotRootParameter[0].InitAsDescriptorTable(1, &cbvTable);
357
358 // A root signature is an array of root parameters.
359 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(1, slotRootParameter, 0, nullptr,
360 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
361
362
363 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
364 ComPtr<ID3DBlob> serializedRootSig = nullptr;
365 ComPtr<ID3DBlob> errorBlob = nullptr;
366 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
367 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
368
369 if(errorBlob != nullptr)
370 {
371 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
372 }
373 ThrowIfFailed(hr);
374
375 ThrowIfFailed(md3dDevice->CreateRootSignature(
376 0,
377 serializedRootSig->GetBufferPointer(),
378 serializedRootSig->GetBufferSize(),
379 IID_PPV_ARGS(&mRootSignature)));
380}

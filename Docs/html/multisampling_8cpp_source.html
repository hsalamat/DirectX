127void TriangleAPP::Draw(const GameTimer& gt)
128{
129 // Reuse the memory associated with command recording.
130 // We can only reset when the associated command lists have finished execution on the GPU.
131 ThrowIfFailed(mDirectCmdListAlloc->Reset());
132
133 //A command list can be reset after it has been added to the command queue via ExecuteCommandList.
134 // Reusing the command list reuses memory.
135 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), mPSO.Get()));
136
137 //You cannot render without a view port.
138 //viewport maps a NDC coordinate (like 0,1) to Render Target (400,0) assuming screen space 800X800 pixels.
139 //also you use viewport to map to subportion of the render target (like top left rectangle of your screen)
140 //also notice viewport(s), because you can have an array of view ports! & is poiting to the address of the first viewport
141 mCommandList->RSSetViewports(1, &mScreenViewport);
142
143 //Each scissor rectangle in the array corresponds to a viewport in an array of viewports
144 mCommandList->RSSetScissorRects(1, &mScissorRect);
145
146 // Indicate a state transition on the resource usage.
147 //Specifies the state of a resource regarding how the resource is being used.
148 //D3D12_RESOURCE_STATE_PRESENT same as D3D12_RESOURCE_STATE_COMMON
149 //Applications should only transition to this state for accessing a resource across different graphics engine types.
150 //Specifically, a resource must be in the COMMON state before being used on a COPY queue
151 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
152 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
153
154
155 // Clear the back buffer and depth buffer.
156 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
157
158 // Specify the buffers we are going to render to.
159 //notice that "targets" not "target" in case we have few passes, filtering, etc.
160 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, nullptr);
161
162 //ID3D12DescriptorHeap* descriptorHeaps[] = { mCbvHeap.Get() };
163 //mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
164
165 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
166
167 mCommandList->IASetVertexBuffers(0, 1, &mVertexBufferView);
168
169 mCommandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
170
171 //there used to a Draw() function where we used to have vertices address as a parameter. PSO took care of this for us. Now we have different PSO for different sets of objects.
172 mCommandList->DrawInstanced(3, 1, 0, 0);
173
174 // Indicate a state transition on the resource usage.
175 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
176 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
177
178 //step4: Done recording commands.
179 ThrowIfFailed(mCommandList->Close());
180
181 // Add the command list to the queue for execution.
182 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
183 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
184
185 // swap the back and front buffers
186 //HRESULT Present(UINT SyncInterval,UINT Flags);
187 //SyncInterval: 0 - The presentation occurs immediately, there is no synchronization.
188 //Flags defined by the DXGI_PRESENT constants. 0: Present a frame from each buffer (starting with the current buffer) to the output.
189 ThrowIfFailed(mSwapChain->Present(0, 0));
190 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
191
192 // Wait until frame commands are complete. This waiting is inefficient and is
193 // done for simplicity. Later we will show how to organize our rendering code
194 // so we do not have to wait per frame.
195 FlushCommandQueue();
196}

200void TriangleAPP::BuildRootSignature()
201{
202 // Shader programs typically require resources as input (constant buffers,
203 // textures, samplers). The root signature defines the resources the shader
204 // programs expect. If we think of the shader programs as a function, and
205 // the input resources as function parameters, then the root signature can be
206 // thought of as defining the function signature.
207
208
210 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(0, nullptr, 0, nullptr, D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
211
212
213 // create a root signature with no slot for now!
214 ComPtr<ID3DBlob> serializedRootSig = nullptr;
215 ComPtr<ID3DBlob> errorBlob = nullptr;
216 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1, &serializedRootSig, &errorBlob);
217
218 if (errorBlob != nullptr)
219 {
220 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
221 }
222 ThrowIfFailed(hr);
223
224 ThrowIfFailed(md3dDevice->CreateRootSignature(
225 0,
226 serializedRootSig->GetBufferPointer(),
227 serializedRootSig->GetBufferSize(),
228 IID_PPV_ARGS(&mRootSignature)));
229}

264

266
267 // Define the geometry for a triangle.
268 Vertex triangleVertices[] =
269 {
270 Vertex({ XMFLOAT3(-0.5f, -0.5f, 0.0f)}),
271 Vertex({ XMFLOAT3(+0.0f, +0.5f, 0.0f) }),
272 Vertex({ XMFLOAT3(+0.5f, -0.5f, 0.0f) }),
273 };
274
275 const UINT vertexBufferSize = sizeof(triangleVertices);
276
277 // Note: using upload heaps to transfer static data like vert buffers is not
278 // recommended. Every time the GPU needs it, the upload heap will be marshalled
279 // over. Please read up on Default Heap usage. An upload heap is used here for
280 // code simplicity and because there are very few verts to actually transfer.
281 md3dDevice->CreateCommittedResource(
282 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_UPLOAD),
283 D3D12_HEAP_FLAG_NONE,
284 &CD3DX12_RESOURCE_DESC::Buffer(vertexBufferSize),
285 D3D12_RESOURCE_STATE_GENERIC_READ,
286 nullptr,
287 IID_PPV_ARGS(&mVertexBuffer));
288
289 // Copy the triangle data to the vertex buffer.
290 UINT8* pVertexDataBegin;
291 CD3DX12_RANGE readRange(0, 0); // We do not intend to read from this resource on the CPU.
292 mVertexBuffer->Map(0, &readRange, reinterpret_cast<void**>(&pVertexDataBegin));
293 memcpy(pVertexDataBegin, triangleVertices, sizeof(triangleVertices));
294 mVertexBuffer->Unmap(0, nullptr);
295
296 //Initialize the vertex buffer view.
297 mVertexBufferView.BufferLocation = mVertexBuffer->GetGPUVirtualAddress();
298 mVertexBufferView.StrideInBytes = sizeof(Vertex);
299 mVertexBufferView.SizeInBytes = vertexBufferSize;

305void TriangleAPP::BuildPSO()
306{
307 D3D12_GRAPHICS_PIPELINE_STATE_DESC psoDesc;
308 ZeroMemory(&psoDesc, sizeof(D3D12_GRAPHICS_PIPELINE_STATE_DESC));
309
310 //bind the input layout to pipeline
311 psoDesc.InputLayout = { mInputLayout.data(), (UINT)mInputLayout.size() };
312 psoDesc.pRootSignature = mRootSignature.Get();
313
314 //in old days, we use to call create vertex shader, now we hook it up to the PSO!
315 psoDesc.VS =
316 {
317 reinterpret_cast<BYTE*>(mvsByteCode->GetBufferPointer()),
318 mvsByteCode->GetBufferSize()
319 };
320 psoDesc.PS =
321 {
322 reinterpret_cast<BYTE*>(mpsByteCode->GetBufferPointer()),
323 mpsByteCode->GetBufferSize()
324 };
325 psoDesc.RasterizerState = CD3DX12_RASTERIZER_DESC(D3D12_DEFAULT);
326 psoDesc.BlendState = CD3DX12_BLEND_DESC(D3D12_DEFAULT);
327 psoDesc.DepthStencilState.DepthEnable = FALSE;
328 psoDesc.DepthStencilState.StencilEnable = FALSE;
329 psoDesc.SampleMask = UINT_MAX;
330 psoDesc.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
331 psoDesc.NumRenderTargets = 1;
332 psoDesc.RTVFormats[0] = mBackBufferFormat; //DXGI_FORMAT_R8G8B8A8_UNORM;
333 //step1
334
335 //We can query the number of quality levels for a given texture format and sample count using the ID3D12Device::CheckFeatureSupport method like so :
336 D3D12_FEATURE_DATA_MULTISAMPLE_QUALITY_LEVELS msQualityLevels;
337 msQualityLevels.Format = mBackBufferFormat;
338 msQualityLevels.SampleCount = 32;
339 msQualityLevels.Flags = D3D12_MULTISAMPLE_QUALITY_LEVELS_FLAG_NONE;
340 msQualityLevels.NumQualityLevels = 0;
341
342 //Note that the second parameter is both an input and output parameter. For the input,
343 //we must specify the texture format, sample count, and flag we want to query
344 //multisampling support for.The function will then fill out the quality level as the output.
345 //Valid quality levels for a texture format and sample count combination range from zero to NumQualityLevels1.
346 //The maximum number of samples that can be taken per pixel is defined by : #define D3D12_MAX_MULTISAMPLE_SAMPLE_COUNT ( 32 )
347
348 ThrowIfFailed(md3dDevice->CheckFeatureSupport(D3D12_FEATURE_MULTISAMPLE_QUALITY_LEVELS, &msQualityLevels, sizeof(msQualityLevels)));
349
350
351 //psoDesc.SampleDesc.Count = 1;
352 m4xMsaaState = true;
353 psoDesc.SampleDesc.Count = m4xMsaaState ? 4 : 1; //The Count member specifies the number of samples to take per pixel
354 psoDesc.SampleDesc.Quality = m4xMsaaState ? (m4xMsaaQuality - 1) : 0; //Quality member is used to specify the desired quality level (what quality level means can vary across hardware manufacturers).
355
356
357 ThrowIfFailed(md3dDevice->CreateGraphicsPipelineState(&psoDesc, IID_PPV_ARGS(&mPSO)));
358}

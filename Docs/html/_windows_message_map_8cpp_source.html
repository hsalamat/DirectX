193std::string WindowsMessageMap::operator()(DWORD msg, LPARAM lp, WPARAM wp) const noexcept
194{
195 constexpr int firstColWidth = 25;
196 const auto i = map.find(msg);
197
198 std::ostringstream oss;
199 if (i != map.end())
200 {
201 oss << std::left << std::setw(firstColWidth) << i->second << std::right;
202 }
203 else
204 {
205 std::ostringstream padss;
206 padss << "Unknown message: 0x" << std::hex << msg;
207 oss << std::left << std::setw(firstColWidth) << padss.str() << std::right;
208 }
209 oss << " LP: 0x" << std::hex << std::setfill('0') << std::setw(8) << lp;
210 oss << " WP: 0x" << std::hex << std::setfill('0') << std::setw(8) << wp << std::endl;
211
212 return oss.str();
213}

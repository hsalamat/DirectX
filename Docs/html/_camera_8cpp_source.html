116void Camera::SetLens(float fovY, float aspect, float zn, float zf)
117{
118 // cache properties
119 mFovY = fovY;
120 mAspect = aspect;
121 mNearZ = zn;
122 mFarZ = zf;
123
124 mNearWindowHeight = 2.0f * mNearZ * tanf( 0.5f*mFovY );
125 mFarWindowHeight = 2.0f * mFarZ * tanf( 0.5f*mFovY );
126
127 XMMATRIX P = XMMatrixPerspectiveFovLH(mFovY, mAspect, mNearZ, mFarZ);
128 XMStoreFloat4x4(&mProj, P);
129}

131void Camera::LookAt(FXMVECTOR pos, FXMVECTOR target, FXMVECTOR worldUp)
132{
133 XMVECTOR L = XMVector3Normalize(XMVectorSubtract(target, pos));
134 XMVECTOR R = XMVector3Normalize(XMVector3Cross(worldUp, L));
135 XMVECTOR U = XMVector3Cross(L, R);

137 XMStoreFloat3(&mPosition, pos);
138 XMStoreFloat3(&mLook, L);
139 XMStoreFloat3(&mRight, R);

251void Camera::UpdateViewMatrix()
252{
253 if(mViewDirty)
254 {
255 XMVECTOR R = XMLoadFloat3(&mRight);
256 XMVECTOR U = XMLoadFloat3(&mUp);
257 XMVECTOR L = XMLoadFloat3(&mLook);
258 XMVECTOR P = XMLoadFloat3(&mPosition);
259
260 // Keep camera's axes orthogonal to each other and of unit length.
261 L = XMVector3Normalize(L);
262 U = XMVector3Normalize(XMVector3Cross(L, R));
263
264 // U, L already ortho-normal, so no need to normalize cross product.
265 R = XMVector3Cross(U, L);
266
267 // Fill in the view matrix entries.
268 float x = -XMVectorGetX(XMVector3Dot(P, R));
269 float y = -XMVectorGetX(XMVector3Dot(P, U));
270 float z = -XMVectorGetX(XMVector3Dot(P, L));
271
272 XMStoreFloat3(&mRight, R);
273 XMStoreFloat3(&mUp, U);
274 XMStoreFloat3(&mLook, L);
275
276 mView(0, 0) = mRight.x;
277 mView(1, 0) = mRight.y;
278 mView(2, 0) = mRight.z;
279 mView(3, 0) = x;
280
281 mView(0, 1) = mUp.x;
282 mView(1, 1) = mUp.y;
283 mView(2, 1) = mUp.z;
284 mView(3, 1) = y;
285
286 mView(0, 2) = mLook.x;
287 mView(1, 2) = mLook.y;
288 mView(2, 2) = mLook.z;
289 mView(3, 2) = z;
290
291 mView(0, 3) = 0.0f;
292 mView(1, 3) = 0.0f;
293 mView(2, 3) = 0.0f;
294 mView(3, 3) = 1.0f;
295
296 mViewDirty = false;
297 }

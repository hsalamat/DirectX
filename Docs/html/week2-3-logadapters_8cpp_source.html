162void InitDirect3DApp::Draw(const GameTimer& gt)
163{
164 // Reuse the memory associated with command recording.
165 // We can only reset when the associated command lists have finished execution on the GPU.
166 ThrowIfFailed(mDirectCmdListAlloc->Reset());
167
168 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
169 // Reusing the command list reuses memory.
170 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
171
172 // Indicate a state transition on the resource usage.
173 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
174 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
175
176 // Set the viewport and scissor rect. This needs to be reset whenever the command list is reset.
177 mCommandList->RSSetViewports(1, &mScreenViewport);
178 mCommandList->RSSetScissorRects(1, &mScissorRect);
179
180 // Clear the back buffer and depth buffer.
181 //mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
182 //Hooman Step1:
183 const float color[] = { 1.0,1.0,0.0,1.0 };
184 const float c = sin(mTimer.TotalTime());
185 const float color2[] = { c,c,1.0,1.0 };
186 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), color2, 0, nullptr);
187
188 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
189
190 // Specify the buffers we are going to render to.
191 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
192
193 // Indicate a state transition on the resource usage.
194 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
195 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
196
197 // Done recording commands.
198 ThrowIfFailed(mCommandList->Close());
199
200 // Add the command list to the queue for execution.
201 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
202 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
203
204 // swap the back and front buffers
205 ThrowIfFailed(mSwapChain->Present(0, 0));
206 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
207
208 // Wait until frame commands are complete. This waiting is inefficient and is
209 // done for simplicity. Later we will show how to organize our rendering code
210 // so we do not have to wait per frame.
211 FlushCommandQueue();
212}

71 //We can query the number of quality levels for a given texture format and sample count using the ID3D12Device::CheckFeatureSupport method like so :
72 D3D12_FEATURE_DATA_MULTISAMPLE_QUALITY_LEVELS msQualityLevels;
73 msQualityLevels.Format = mBackBufferFormat;
74 msQualityLevels.SampleCount = 4;
75 msQualityLevels.Flags = D3D12_MULTISAMPLE_QUALITY_LEVELS_FLAG_NONE;
76 msQualityLevels.NumQualityLevels = 0;
77
78 //Note that the second parameter is both an input and output parameter. For the input,
79 //we must specify the texture format, sample count, and flag we want to query
80 //multisampling support for.The function will then fill out the quality level as the output.
81 //Valid quality levels for a texture format and sample count combination range from zero to NumQualityLevels–1.

84 ThrowIfFailed(md3dDevice->CheckFeatureSupport(D3D12_FEATURE_MULTISAMPLE_QUALITY_LEVELS, &msQualityLevels, sizeof(msQualityLevels)));

101void InitDirect3DApp::Draw(const GameTimer& gt)
102{
103 // Reuse the memory associated with command recording.
104 // We can only reset when the associated command lists have finished execution on the GPU.
105 ThrowIfFailed(mDirectCmdListAlloc->Reset());
106
107 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
108 // Reusing the command list reuses memory.
109 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
110
111 // Indicate a state transition on the resource usage.
112 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
113 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
114
115 // Set the viewport and scissor rect. This needs to be reset whenever the command list is reset.
116 mCommandList->RSSetViewports(1, &mScreenViewport);
117 mCommandList->RSSetScissorRects(1, &mScissorRect);
118
119 // Clear the back buffer and depth buffer.
120 //mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
121 //Hooman Step1:
122 const float color[] = { 1.0,1.0,0.0,1.0 };
123 const float c = sin(mTimer.TotalTime());
124 const float color2[] = { c,c,1.0,1.0 };
125 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), color2, 0, nullptr);
126
127 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
128
129 // Specify the buffers we are going to render to.
130 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
131
132 // Indicate a state transition on the resource usage.
133 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
134 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
135
136 // Done recording commands.
137 ThrowIfFailed(mCommandList->Close());
138
139 // Add the command list to the queue for execution.
140 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
141 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
142
143 // swap the back and front buffers
144 ThrowIfFailed(mSwapChain->Present(0, 0));
145 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
146
147 // Wait until frame commands are complete. This waiting is inefficient and is
148 // done for simplicity. Later we will show how to organize our rendering code
149 // so we do not have to wait per frame.
150 FlushCommandQueue();
151}
